SETS
>>> s={2,8,6,4,5,3,2,1}
>>> m={7,8,5,4,1,2,6,9}
>>> s|m
{1, 2, 3, 4, 5, 6, 7, 8, 9}
>>> s&m
{1, 2, 4, 5, 6, 8}
>>> s-m
{3}
